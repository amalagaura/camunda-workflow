def initialize
  @engine_url = 'http://localhost:8080'
  @engine_route_prefix = 'rest-engine'
  @camunda_user = ''
  @camunda_password = ''
  @worker_id = '0'
  @lock_duration = 14.days
  @max_polling_tasks = 2
  @long_polling_duration = 30.seconds
  @tenant_id = if defined?(Rails)
                 Rails.env.test? ? 'test-environment' : nil
               end
end
